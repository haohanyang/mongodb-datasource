// Global variables
// Reference: https://docs.aws.amazon.com/grafana/latest/userguide/variables-types.html

/**
 * This variable is the name of the current dashboard.
 */
declare const $__dashboard: string;

/**
 * Beginning of time range in Unix millisecond epoch, e.g., 1594671549254
 */
declare const $__from: number;

/**
 * End of time range in Unix millisecond epoch, e.g., 1594671549254
 */
declare const $__to: number;


/**
 * You can use the `$__interval` variable as a parameter to group by time (for InfluxDB, Myself, Postgres, MSSQL), Date histogram interval (for OpenSearch), or as a summarize function parameter (for Graphite).
 * 
 * The Grafana workspace automatically calculates an interval that can be used to group by time in queries. When there are more data points than can be shown on a graph, queries can be made more efficient by grouping by a larger interval. For example, it is more efficient to group by 1 day than by 10s when looking at 3 months of data. The graph will look the same, and the query will be faster. The `$__interval` is calculated by using the time range and the width of the graph (the number of pixels).
 * 
 * Approximate Calculation: `(from - to) / resolution`
 */
declare const $__interval: string;

/**
 * This variable is the `$__interval` variable in milliseconds, not a time interval formatted string. For example, if the `$__interval` is 20m then the `$__interval_ms` is `1200000`.
 */
declare const $__interval_ms: number;

/**
 * This variable is the ID of the current organization. The variable `${__org.name}` is the name of the current organization.
 */
declare const $__org: string

/**
 * The variable `${__user.id}` is the ID of the current user. The variable `${__user.login}` is the login handle of the current user. The variable `${__user.email}` is the email for the current user.
 */
declare const $__user: string

declare const $__local_from: number;

declare const $__local_to: number;

/**

declare const $__from_oid: string;

/**

declare const $__to_oid: string;

// MongoDB constructors
// Reference: https://www.mongodb.com/docs/manual/reference/method/js-constructor/

declare class ObjectId {
    /**
     * Returns a new ObjectId.
     * @param value Optional. A 24 character hexadecimal string value for the new ObjectId, or an integer value, in seconds, is added to the Unix epoch to create the new timestamp.
     */
    constructor(value?: string | number)
};

/**
 * Returns a new ObjectId.
 * @param value Optional. A 24 character hexadecimal string value for the new ObjectId, or an integer value, in seconds, is added to the Unix epoch to create the new timestamp.
 */
declare function ObjectId(value?: string | number): ObjectId;

/**
 * Returns a date either as a string or as a Date object. The date can contain a date and a time, known as a datetime.
 * @param value Optional. An ISO-8601 date string with a year within the inclusive range 0 through 9999. If not provided, it returns the current date.
 */
declare function ISODate(value?: string): Date;


declare class UUID {
    /**
     * Generates a BSON UUID object.
     * @param hex Optional. Specify a 36 character string to convert to a UUID BSON object. If not provided, MongoDB generates a random UUID in RFC 4122 v4 format. 
     */
    constructor(hex?: string)
};

/**
 * Generates a BSON UUID object.
 * @param hex Optional. Specify a 36 character string to convert to a UUID BSON object. If not provided, MongoDB generates a random UUID in RFC 4122 v4 format. 
 */
declare function UUID(hex?: string): UUID;
